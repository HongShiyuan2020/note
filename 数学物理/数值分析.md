### 迭代法

$$ 
\begin{split}
Ax = b &\iff (M − N)x = b \\ 
&\iff Mx = Nx + b \\ 
&\iff x = M^{−1}Nx + M^{-1}b \\
& \iff Bx + g
\end{split}
$$

#### 收敛性

$$
\begin{split}
	& \begin{cases}
	x^{k+1} &= Bx^{k} + g & (1)\\
	x^{*} &= Bx^* + g & (2)\\ 
	\end{cases} \\
	&\\
	&(1) - (2) = x^{k+1}-x^{*} = B(x^{k} - x^{*}) \\
	&\text{let} \quad e^{k} = x^k - x^* \\\\
	&\implies 
	e^{k+1} = Be^k = B^2e^{k-1} = \dots = B^{k+1}e^0 = 0 \quad\text{时收敛} \\ 
	&\iff \lim_{k \to \infty} B^k = 0 \\
	&\iff \rho(B) < 1 \quad\rho(B)\text{为矩阵的谱半径} \\
	&\rho(B) = \max_{1 \leq i \leq n}\{ | \lambda_{i}(B) | \}
\end{split}
$$

#### Jacobi迭代

$$
\begin{split}
& A = D − L − R, D_{ii} \ne 0 \\
& D: 对角矩阵 \\
& L: 下三角矩阵 \\
& R: 上三角矩阵 \\
\\\\
& Ax=b \\
&\iff (D-L-R)x=b \\
&\iff Dx = (L+R)x + b \\
&\iff x = D^{-1}(L+R)x+D^{-1}b \\
&\iff x^{(k+1)} = B_{J}x^{(k)} + g_{J}
\end{split} 
$$

##### 收敛性

$$
\begin{split}
&充要条件: \rho(B_J) < 1 \\
&充分条件：\|B_J\| < 1 \\
\end{split}
$$

#### Gauss-Seidel 迭代

$$
\begin{split}
& Ax = (D-L-R)x = b \\
&\iff (D-L)x = Rx + b \\
&\iff x = (D-L)^{-1}Rx + (D-L)^{-1}b \\
&\iff x = B_{G}x + g_{G} \\
&\iff x^{(k+1)} = B_Gx^{(k)} + g_G \\
\end{split}
$$

##### 收敛条件

$$
\begin{split}
&充要条件: \rho({B_G}) < 1 \\
&充分条件: \|B_G\| < 1 \\
\end{split}
$$

### 插值

#### Lagrange 插值法

$$ 
\begin{split}
L_n(x) &= \sum_{i=0}^{n} y_il_i(x) \\
l_i(x) &= \frac{(x-x_0)(x-x_1)\dots(x-x_{i-1})(x-x_{i+1})\dots(x-x_n)}{(x_i-x_0)(x_i-x_1)\dots(x_i-x_{i-1})(x_i-x_{i+1})\dots(x_i-x_n)}
\end{split}
$$

\xi \in (a, b), 使得: \\\\
&f(x) - L_n(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!}\omega_{n+1}(x) \\ 
&\omega_{n+1}(x) = \prod_{i=0}^{n}(x-x_i)
\end{split}

#### Hermite 插值

##### 目标

$$
\begin{split}
已知节点函数值和导数值: \\
&(x_0, y_0), (x_1, y_1), · · · , (x_n, y_n) \\
&(x_0, y_0^{'}), (x_1, y_1^{'}), · · · , (x_n, y_n^{'}) \\\\
试确定一个2n + 1 次多项式: \\
&P_{2n+1}(x) = a_0 + a_1x + a_2x^2 + \dots + a_{2n+1}x^{2n+1}\\
满足：\\
&\begin{cases}
P_{2n+1}(x_i) &= y_i \\
P^{'}_{2n+1}(x_i) &= y_i^{'} \\
\end{cases}
\end{split}
$$

##### 方法

$$
\begin{split}
H_{2n+1}(x) &= \sum_{i=0}^{n} a_ix^i \\
&=\sum_{i=0}^n (y_i\lambda_i(x) + y^{'}_i\mu_i({x})) \\
满足: \\
&\begin{cases}
\lambda_i(x_j) = \delta_{ij}, &\lambda^{'}_i(x_j)=0 \\
\mu_i(x_j) = 0, &\mu^{'}_i(x_j) = \delta_{ij} \\ 
\end{cases}\\
那么: \\
&\mu_i(x) = (x-x_i)l_i(x)^2 \\
&\lambda_i(x) = [1-2(x-x_i)l^{'}_i(x)]l_i(x)^2 \\
\end{split}
$$

##### 插值误差：

$$
f(x)-H_{2n+1}(x) = \frac{f^{(2n+2)}(\xi)}{(2n+2)!}\omega^{2}_{2n+2}(x)
$$

## 积分变换

### Fourier 变换

$$
F(\omega) = \int_{-\infty}^{\infty} f(t)e^{-i\omega t} dt
$$

#### 逆变换

$$
f(t) = \frac{1}{2\pi} \int_{-\infty}^{+\infty} F(\omega)e^{i\omega t} d\omega 
$$

#### 正弦与余弦变换

$$
\begin{split}
&F_s(\omega) = \int_0^{+\infty} f_s(t)\sin(\omega t) dt \\ 
&f_s(t) = \frac{2}{\pi} \int_0^{+\infty} F_s(\omega)\sin(\omega t) d\omega \\
&F_c(\omega) = \int_0^{+\infty} f_c(t)\cos(\omega t) dt \\
&f_c(t) = \frac{2}{\pi} \int_0^{+\infty} F_c(\omega) \cos(\omega t) d\omega \\
\end{split}
$$
#### 性质

$$
\begin{split}
线性性: \\
&\mathcal{F}(c_1f(t)+c_2g(t)) = c_1\mathcal{F}(f(t)) + c_2\mathcal{F(g(t))} \\
位移性: \\
&\mathcal{F}(f(t\pm t_0)) = e^{i\omega \pm t0}\mathcal{F}(f(t)) \\
尺度变换性: \\
&\mathcal{F(f(kt))} = \frac{1}{|k|}\mathcal{F}(\frac{\omega}{k}) \\
对称性: \\
&\mathcal{F}(F(t)) = 2\pi f(-\omega) \\
微分性: \\
&\mathcal{F}(f^{(n)}) = (i\omega)^nF(\omega) \\
卷积性: \\
&卷积定义: \int_{-\infty}^{\infty} f(\tau)g(t-\tau) \space d\tau = f(t) * g(t) \\
&\mathcal{F}(f(t)*g(t)) = \mathcal{F}(f(t))\mathcal{F}(g(t)) 
\end{split}
$$

### Laplace 变换
$$
F(s) = \int_0^{\infty} f(t)e^{-st} dt
$$
#### 常见函数的Laplace变换

$$
\begin{split}
\mathcal{L}(1) &= \int_0^{\infty} e^{-st} \space dt = \frac{1}{s}, s > 0 \\
\mathcal{L}(e^{at}) &= \int_0^{\infty} e^{at} e^{-st} \space dt = \frac{1}{s-a}, s > a \\
\mathcal{L}(\sin(at)) &= \int_0^{\infty} \sin(at) e^{-st} dt = \frac{a}{s^2+a^2}, a > 0 \\
\mathcal{L}(\cos(at)) &= \int_0^{\infty} \cos(at) e^{-st} dt = \frac{s}{s^2+a^2}, s > 0 \\
\mathcal{L}(t^n) &= \int_0^{\infty} t^{n} e^{-st} dt = \frac{n!}{s^{n+1}}, s > 0 \\
\end{split}
$$

#### 性质

$$
\begin{split}
线性性: \\
& \mathcal{L}(c_1f(t)+c_2g(t)) = c_1\mathcal{L}(f(t)) + c_2\mathcal{L}(g(t)) \\
微分性: \\
& \mathcal{L}(f^{'}(t)) = sF(s) - f(0) \\
& \mathcal{L}(f^{'}(t)) = s^nF(s) - \sum_{i = 0} s^{n-i-1}f^{(i)}(0) \\
积分性: \\
&\mathcal{L}(\int_0^t f(\tau) d\tau) = \frac{1}{s} F(s)\\
\end{split}
$$
